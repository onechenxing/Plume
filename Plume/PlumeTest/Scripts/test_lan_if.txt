	if n == 1 {
		return 1
	}elif n >= 10{
		return 10
	}elif n < 0{
		return -1
	}else{
		return 0
	}
}

printu:(Check:1),'1'
printu:(Check:10),'10'
printu:(Check:20),'10'
printu:(Check:-1),'-1'
printu:(Check:5),'0'
